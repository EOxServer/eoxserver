41      This function handles all incoming OWS requests. It configures basic

45      @param  request     A django.http.HttpRequest object containing the

54      if request.method == 'GET':

60          ows_req = EOxSOWSRequest(

64      else: 
65          raise Exception("Unsupported request method '%s'" % request.method)

67      logging.basicConfig(
